CREATE DATABASE lankacapital;

USE lankacapital;

CREATE TABLE Employee (
    Emp_ID INT PRIMARY KEY,
    Emp_FirstName VARCHAR(50) NOT NULL,
    Emp_LastName VARCHAR(50) NOT NULL,
    Emp_Email VARCHAR(50) NOT NULL,
    Emp_Role ENUM('fo','admin','receptionist','user','customer') DEFAULT 'user',
    Emp_Address VARCHAR(150) NOT NULL,
    Emp_BasicSalary DECIMAL(10,2) NOT NULL,
    Emp_Password VARCHAR(255) NOT NULL DEFAULT '1234'
);

CREATE TABLE Emp_Tele (
    Emp_ID INT,
    Emp_Tele VARCHAR(12),
    PRIMARY KEY (Emp_ID, Emp_Tele),
    FOREIGN KEY (Emp_ID) REFERENCES Employee(Emp_ID)
        ON DELETE CASCADE
);

CREATE TABLE Customer (
    Customer_ID DOUBLE PRIMARY KEY,
    Customer_Name VARCHAR(255) NOT NULL,
    Customer_Email VARCHAR(50) NOT NULL,
    Customer_Address VARCHAR(255) NOT NULL,
    Customer_Role VARCHAR(50) NOT NULL DEFAULT 'customer',
    Customer_Password VARCHAR(255) NOT NULL DEFAULT '1234'
);

CREATE TABLE Customer_Tele (
    Customer_ID DOUBLE,
    Customer_Tele VARCHAR(12),
    PRIMARY KEY (Customer_ID, Customer_Tele),
    FOREIGN KEY (Customer_ID) REFERENCES Customer(Customer_ID)
        ON DELETE CASCADE
);

CREATE TABLE Loan (
    File_ID INT AUTO_INCREMENT PRIMARY KEY,
    File_No VARCHAR(255) UNIQUE,
    Customer_ID DOUBLE NOT NULL,
    Loan_Amount DECIMAL(10,2) NOT NULL,
    Loan_Installments INT NOT NULL,
    Loan_Interest_Rate DECIMAL(5,2) NOT NULL,
    Loan_Created_Date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (Customer_ID) REFERENCES Customer(Customer_ID)
        ON DELETE CASCADE
);

CREATE VIEW Accounts AS
SELECT 
    Emp_Email AS username,
    Emp_Password AS password
FROM Employee

UNION

SELECT 
    Customer_Email AS username,
    Customer_Password AS password
FROM Customer;

DELIMITER $$
CREATE TRIGGER before_insert_loan
BEFORE INSERT ON Loan
FOR EACH ROW
BEGIN
    DECLARE next_id INT;
    SELECT AUTO_INCREMENT 
    INTO next_id
    FROM information_schema.TABLES
    WHERE TABLE_NAME = 'Loan'
    AND TABLE_SCHEMA = DATABASE();

    SET NEW.File_No = CONCAT('D', LPAD(next_id, 3, '0'));
END$$
DELIMITER ;
